    Chat Parts Reference

    All the possbile data to query from a chat part.

    subscription{
  chatMessage(channelId:6) {
    message,
    user {
      id
    },
    tokens {
        text,
        type,
      ...on EmoteToken {
        src,
        text,
        type,
      },
      ...on TextToken {
        text,
        type
      },
      ...on UrlToken {
        text,
        type,
        url
      }
    }
  }
}
